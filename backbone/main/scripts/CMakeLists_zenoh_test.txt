cmake_minimum_required(VERSION 3.10)
project(zenoh_cpp_shm_test CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find Zenoh C++
find_package(zenohc REQUIRED)

# Add executable
add_executable(zenoh_cpp_shm_test zenoh_cpp_shm_test.cpp)

# Link libraries
target_link_libraries(zenoh_cpp_shm_test
    PRIVATE
    ${zenohc_LIBRARIES}
)

# Include directories
target_include_directories(zenoh_cpp_shm_test
    PRIVATE
    ${zenohc_INCLUDE_DIRS}
)

# Compiler flags
target_compile_options(zenoh_cpp_shm_test PRIVATE -Wall -Wextra)
